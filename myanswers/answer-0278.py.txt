import pandas as pd
import numpy as np


def reconstruir_velocidad_faltante(df):

    df_corregido = df.copy()

    velocidad_estimada = df_corregido["posicion"].diff()
    velocidad_estimada.iloc[0] = 0

    velocidad_original = df_corregido["velocidad"].copy()

    df_corregido["velocidad"] = df_corregido["velocidad"].fillna(
        velocidad_estimada
    )

    error = np.abs(velocidad_estimada - velocidad_original)

    error = error.fillna(0)

    umbral = error.mean()

    indices_error = np.where(error > umbral)[0]

    return {
        "df_corregido": df_corregido,
        "indices_error": indices_error
    }
